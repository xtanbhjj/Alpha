A8255_B EQU 0602H

	DB 06H
	DB 5BH
	DB 4FH
	DB 66H
	DB 6DH
	DB 7DH
	DB 07H
	DB 7FH
	DB 6FH

CODE SEGMENT
	ASSUME CS:CODE,DS:DATA

START:
	MOV DX,A8255_CON
	MOV AL,81H
	OUT DX,AL
	MOV AX,DATA
	MOV DS,AX
	
	LEA SI,TABLE1
	MOV CX,0AH
	MOV BX,0000H
X1:
	PUSH CX
	MOV CX,06H
	MOV AL,11011111B
X2:
	MOV DX,A8255_A
	OUT DX,AL
	ROR AL,1

	MOV AL,[BX+SI]
	MOV DX,A8255_B
	OUT DX,AL
	POP AX
	CALL DELAY
	LOOP X2
	POP CX
	INC BX
	LOOP X1
	JMP START
DELAY:
	PUSH CX
	MOV CX,0FFFFH
X4:
	LOOP X4
	MOV CX,0FFFFH
X5:
	LOOP X5
	POP CX
	RET
